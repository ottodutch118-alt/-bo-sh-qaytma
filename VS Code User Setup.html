4️⃣ app.js

JavaScript
let loads = [];

function addLoad(){

let from = document.getElementById("from").value;
let to = document.getElementById("to").value;
let type = document.getElementById("type").value;
let price = document.getElementById("price").value;

let load = {
from: from,
to: to,
type: type,
price: price
};

loads.push(load);

showLoads();
}

function showLoads(){

let list = document.getElementById("loads");
list.innerHTML = "";

loads.forEach((l,index)=>{

let item = document.createElement("li");

item.innerHTML =
"📦 " + l.type +
"<br>📍 " + l.from +
" → " + l.to +
"<br>💰 " + l.price +
"<br><button onclick='deleteLoad("+index+")'>O‘chirish</button>";

list.appendChild(item);

});
}

function deleteLoad(index){
loads.splice(index,1);
showLoads();
}
